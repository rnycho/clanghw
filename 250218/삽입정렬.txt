#include <stdio.h>

void main()
{
    int arr[] = {8, 3, 5, 2, 7};
    printf("초 기 : 8 3 5 2 7\n");
	
	for (int i = 1; i < 5; i++) {
		int temp = arr[i];
		int j = i - 1;
		
		while (j >= 0 && arr[j] > temp) {
			arr[j + 1] = arr[j];
			j--;
		}
		arr[j + 1] = temp;
		printf("%d회차 : ", i);
		
		for (int i = 0; i < 5; i++) {
			printf("%d ", arr[i]);
		}
		printf("\n");
	}
}
